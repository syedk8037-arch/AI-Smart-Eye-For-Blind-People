package com.example.smartglass

import android.Manifest
import android.content.Intent
import android.content.pm.PackageManager
import android.os.Bundle
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.ActivityCompat
import androidx.core.content.ContextCompat

class MainActivity : AppCompatActivity() {

    private lateinit var bluetoothHelper: BluetoothHelper

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btnAbout: Button = findViewById(R.id.btnAbout)
        val btnFaces: Button = findViewById(R.id.btnFaces)
        val btnSync: Button = findViewById(R.id.btnSync)
        val btnCaptured: Button = findViewById(R.id.btnCaptured)
        //val secretBtn = findViewById<Button>(R.id.btnSecretVideo)// ⬅ new button

        // 🔹 Navigation
        btnAbout.setOnClickListener {
            startActivity(Intent(this, AboutActivity::class.java))
        }

        btnFaces.setOnClickListener {
            // your existing faces screen
            startActivity(Intent(this, FacesActivity::class.java))
        }

        btnSync.setOnClickListener {
            startActivity(Intent(this, SyncActivity::class.java))
        }

        btnCaptured.setOnClickListener {
            // new captured-images screen
            startActivity(Intent(this, CapturedImagesActivity::class.java))
        }

        val btnCapture: Button = findViewById(R.id.btnCapture)
        btnCapture.setOnClickListener {
            startActivity(Intent(this, CaptureActivity::class.java))
        }

        //secretBtn.setOnClickListener {
       //     val i = Intent(this, SecretVideoActivity::class.java)
        //    startActivity(i)
    //    }


        // 🔹 Bluetooth permission request (unchanged)
        if (ContextCompat.checkSelfPermission(
                this,
                Manifest.permission.BLUETOOTH_CONNECT
            ) != PackageManager.PERMISSION_GRANTED
        ) {
            ActivityCompat.requestPermissions(
                this,
                arrayOf(Manifest.permission.BLUETOOTH_CONNECT),
                101
            )
        } else {
            connectToPi()
        }
    }

    private fun connectToPi() {
        bluetoothHelper = BluetoothHelper("SmartEyePi")

        Thread {
            val connected = bluetoothHelper.connect()
            runOnUiThread {
                if (connected)
                    Toast.makeText(this, "Connected ✓", Toast.LENGTH_SHORT).show()
                else
                    Toast.makeText(this, "Connection failed ✗", Toast.LENGTH_SHORT).show()
            }
        }.start()
    }

    override fun onRequestPermissionsResult(
        requestCode: Int,
        permissions: Array<out String>,
        grantResults: IntArray
    ) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults)
        if (requestCode == 101 && grantResults.isNotEmpty() &&
            grantResults[0] == PackageManager.PERMISSION_GRANTED
        ) {
            connectToPi()
        }
    }

    override fun onDestroy() {
        super.onDestroy()
        if (::bluetoothHelper.isInitialized) {
            bluetoothHelper.disconnect()
        }
    }
}
